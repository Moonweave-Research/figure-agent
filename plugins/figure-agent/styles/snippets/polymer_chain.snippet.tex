%% Sulfur-rich copolymer chain with monomer-resolved schematic backbone.

%%
%% Usage:
%%   \input{styles/snippets/polymer_chain.snippet.tex}
%%   \PolymerChain{x_anchor}{y_anchor}{monomer_count}{s_csv}
%%
%% Args:
%%   #1 = anchor_x (TikZ cm)
%%   #2 = anchor_y (TikZ cm)
%%   #3 = monomer_count (integer 4..16)
%%   #4 = s_csv = comma-separated 1-based monomer indices that carry a single -S branch
%%
%% Geometry contract:
%%   - Backbone is a light gray monomer-resolved zigzag path
%%   - Each monomer vertex gets a tiny node marker
%%   - S branches alternate above/below the backbone to avoid a comb-like structure
%%   - Caller anchors at the LEFT end; chain extends right
%%   - Approx horizontal span = (monomer_count - 1) * 0.42 cm
%%
%% Density encoding (advisor-validated):
%%   sparse  = S indices spaced ~4 apart (e.g. 3,7,11)
%%   rich    = S indices contiguous     (e.g. 6,7,8,9)
%%   mixed   = consumer composes (e.g. 3,7,8,9 = sparse left + rich middle)

% Dependency: xstring — loaded by polymer-paper-preamble.sty.
% Snippet is intended to be \input-ed in the document preamble (before \begin{document}).

\makeatletter
\newcommand{\PolymerChain}[4]{%
  \begingroup
  \def\PCsidecsv{,#4,}%
  \pgfmathtruncatemacro{\PCnmono}{#3}%
  \def\PCstep{0.42}%
  \def\PCamp{0.115}%
  \def\PCbranch{0.24}%
  % Backbone segments: monomer-resolved, but schematic rather than full structure.
  \foreach \PCi in {2,...,\PCnmono} {%
    \pgfmathtruncatemacro{\PCprev}{\PCi-1}%
    \pgfmathsetmacro{\PCxprev}{(#1)+(\PCprev-1)*\PCstep}%
    \pgfmathsetmacro{\PCyprev}{(#2)+ifthenelse(mod(\PCprev,2)==0,\PCamp,-\PCamp)}%
    \pgfmathsetmacro{\PCx}{(#1)+(\PCi-1)*\PCstep}%
    \pgfmathsetmacro{\PCy}{(#2)+ifthenelse(mod(\PCi,2)==0,\PCamp,-\PCamp)}%
    \draw[cGray!85!black, line width=0.56pt, line cap=round, line join=round]
      (\PCxprev,\PCyprev) -- (\PCx,\PCy);
  }%
  % Monomer vertices + sulfur side groups.
  \foreach \PCi in {1,...,\PCnmono} {%
    \pgfmathsetmacro{\PCx}{(#1)+(\PCi-1)*\PCstep}%
    \pgfmathsetmacro{\PCy}{(#2)+ifthenelse(mod(\PCi,2)==0,\PCamp,-\PCamp)}%
    \draw[cGray!70, fill=white, line width=0.28pt] (\PCx,\PCy) circle (0.026);
    \edef\PCneedle{,\PCi,}%
    \IfSubStr{\PCsidecsv}{\PCneedle}%
      {%
        \pgfmathsetmacro{\PCsign}{ifthenelse(mod(\PCi,2)==0,1,-1)}%
        \pgfmathsetmacro{\PCsx}{\PCx+0.03*\PCsign}%
        \pgfmathsetmacro{\PCsy}{\PCy+\PCsign*\PCbranch}%
        \draw[cAmber, line width=0.36pt, line cap=round]
          (\PCx,\PCy) -- (\PCsx,\PCsy-0.05*\PCsign);
        \node[
          circle,
          draw=cAmber,
          fill=cAmber!12,
          inner sep=0pt,
          minimum size=1.05mm,
          line width=0.28pt
        ] at (\PCsx,\PCsy) {};
        \node[
          text=cAmber,
          font=\sffamily\bfseries\fontsize{3.6}{4.2}\selectfont
        ] at (\PCsx,\PCsy) {S};
      }%
      {}%
  }%
  \endgroup
}
\makeatother
