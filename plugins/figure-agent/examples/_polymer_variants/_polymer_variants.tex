% Helper: \DIBblock{x_center}{y_center}
\newcommand{\DIBblock}[2]{%
  \begin{scope}[shift={(#1,#2)}]
    \def\bR{0.18}%
    \foreach \k in {0,...,5} {%
      \pgfmathsetmacro{\bxa}{\bR*cos(\k*60)}%
      \pgfmathsetmacro{\bya}{\bR*sin(\k*60)}%
      \pgfmathsetmacro{\bxb}{\bR*cos((\k+1)*60)}%
      \pgfmathsetmacro{\byb}{\bR*sin((\k+1)*60)}%
      \draw[black!85, line width=0.45pt] (\bxa,\bya) -- (\bxb,\byb);
    }%
    \foreach \k in {0,...,5} {%
      \pgfmathsetmacro{\bxa}{\bR*cos(\k*60)}%
      \pgfmathsetmacro{\bya}{\bR*sin(\k*60)}%
      \node[circle, fill=black!85, draw=black, line width=0.18pt,
            inner sep=0pt, minimum size=1.35mm] at (\bxa,\bya) {};
    }%
    \def\pOff{0.16}%
    \draw[black!85, line width=0.42pt]
      (-\bR,0) -- (-\bR-\pOff, 0);
    \node[circle, fill=black!85, draw=black, line width=0.18pt,
          inner sep=0pt, minimum size=1.2mm] at (-\bR-\pOff, 0) {};
    \draw[black!85, line width=0.42pt]
      (\bR,0) -- (\bR+\pOff, 0);
    \node[circle, fill=black!85, draw=black, line width=0.18pt,
          inner sep=0pt, minimum size=1.2mm] at (\bR+\pOff, 0) {};
  \end{scope}
}

% Helper: \SrunFrom{x_start}{y}{n_atoms} — draws S_n starting just right of x_start
\newcommand{\SrunFrom}[3]{%
  \begin{scope}
    \foreach \i in {1,...,#3} {%
      \pgfmathsetmacro{\sx}{(#1)+\i*0.22}%
      \draw[cAmber, line width=1.1pt, line cap=round]
        (\sx-0.22,#2) -- (\sx,#2);
      \node[circle, fill=cAmber, draw=cAmber!75!black, line width=0.30pt,
            inner sep=0pt, minimum size=1.55mm] at (\sx,#2) {};
    }%
  \end{scope}
}

% Helper: \BlackBall4{x_start}{y} — 4 black balls in a row, bonds drawn
\newcommand{\BlackRow}[3]{%
  % #1 = x_start (first ball center), #2 = y, #3 = number of balls
  \begin{scope}
    \foreach \i in {1,...,#3} {%
      \pgfmathsetmacro{\bx}{(#1)+(\i-1)*0.22}%
      \ifnum\i>1
        \pgfmathsetmacro{\bxprev}{(#1)+(\i-2)*0.22}%
        \draw[black!80, line width=0.40pt] (\bxprev,#2) -- (\bx,#2);
      \fi
      \node[circle, fill=black!85, draw=black, line width=0.18pt,
            inner sep=0pt, minimum size=1.30mm] at (\bx,#2) {};
    }%
  \end{scope}
}
