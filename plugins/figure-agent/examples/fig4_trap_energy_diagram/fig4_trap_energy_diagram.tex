% fig4_trap_energy_diagram — bimodal trap-state energy landscape (schematic).
% figure-agent figure #4 (3rd diagnostic E2E run, 2026-06-20). SCHEMATIC ONLY.
% Left: energy-level diagram (Ec/Ev, shallow tail band, deep S-S-radical levels).
% Right: bimodal trap DOS g(Et) on the same energy axis (peaks aligned to levels).
% Relational invariants (lever-B): shallow ABOVE deep; Ec ABOVE Ev; DOS bimodal,
% peaks aligned; shallow blue/teal, deep red.

\usepackage{polymer-paper-preamble}

\begin{document}
\resizebox{178mm}{!}{%
\begin{tikzpicture}[
  every node/.style={font=\sffamily\fontsize{7}{8.4}\selectfont},
  bandSolid/.style={line width=1.2pt, cGray!82!black},
  vacDash/.style={line width=0.8pt, cGray!60!black, dash pattern=on 3pt off 2pt},
  deepLevel/.style={line width=1.4pt, cRed!75!black},
  electron/.style={circle, fill=cBlue!70, draw=cBlue!55!black, line width=0.3pt,

  capArr/.style={-{Stealth[length=5pt,width=3.5pt]}, cGray!70!black, line width=0.9pt},
  relArr/.style={-{Stealth[length=5pt,width=3.5pt]}, cGray!65!black, line width=0.8pt,
                 dash pattern=on 2.5pt off 1.6pt},
  gapArr/.style={{Stealth[length=5pt,width=3.5pt]}-{Stealth[length=5pt,width=3.5pt]},

  axisArr/.style={-{Stealth[length=4pt,width=3pt]}, cGray!70!black, line width=0.7pt},
  dosCurve/.style={line width=1.5pt, cAmber!85!black},
  connector/.style={line width=0.5pt, cGray!55, dash pattern=on 2pt off 1.6pt},
  labelStd/.style={font=\sffamily\fontsize{7}{8.4}\selectfont},
  labelMute/.style={font=\sffamily\itshape\fontsize{7}{8.4}\selectfont, text=cGray!80!black},
  labelBack/.style={fill=white, fill opacity=0.92, text opacity=1, inner sep=1.0pt},
  panelLetter/.style={font=\sffamily\bfseries\fontsize{8}{9.6}\selectfont, text=cGray!90!black},
]

% ===================== PANEL A — energy-level diagram =====================
\node[panelLetter] at (0.55, 9.45) {a};
\node[labelStd, font=\sffamily\bfseries, anchor=west] at (0.95, 9.45) {Trap-state energy landscape};

% energy axis
\draw[axisArr] (2.00,1.35) -- (2.00,9.10);
\node[labelMute, anchor=south, rotate=90] at (1.70,5.3) {Energy};

% extended (conduction) states fill + valence band fill
\fill[cBlue!8] (2.40,7.20) rectangle (7.40,8.45);
\fill[cBlue!8] (2.40,1.70) rectangle (7.40,2.90);

% levels
\draw[vacDash] (2.40,8.90) -- (7.40,8.90);
\node[labelMute, anchor=west] at (7.55,8.90) {vacuum level};
\draw[bandSolid] (2.40,7.20) -- (7.40,7.20);
\node[labelStd, anchor=west] at (7.55,7.20) {mobility edge $E_c$};

\node[labelStd, anchor=west] at (7.55,2.90) {valence band $E_v$};
\node[labelMute, anchor=west] at (2.55,8.05) {extended states};

% band gap double arrow
\draw[gapArr] (7.05,2.98) -- (7.05,7.12);
\node[labelMute, labelBack, anchor=east] at (6.90,5.05) {$E_g \approx 2$ eV};

\node[labelStd, labelBack, text=cTeal!50!black, anchor=west] at (7.28,6.86) {shallow traps (tail)};
\node[labelMute, labelBack, anchor=west] at (7.28,6.55) {$\sim$0.1--0.3 eV};

\node[panelLetter] at (11.05, 9.45) {b};
\node[labelStd, font=\sffamily\bfseries, anchor=west] at (11.45, 9.45) {Trap density of states};

\draw[axisArr] (11.60,2.40) -- (16.40,2.40);

  plot[smooth, tension=0.7] coordinates {

    (12.10,5.85) (12.70,6.35) (13.65,6.74) (12.35,7.10) (11.70,7.25)};
\node[labelStd, text=cRed!70!black, anchor=west] at (14.30,4.58) {deep};
\node[labelStd, text=cTeal!50!black, anchor=west] at (13.80,6.80) {shallow};

% connectors tying panel-a levels to panel-b peaks (same energy)
