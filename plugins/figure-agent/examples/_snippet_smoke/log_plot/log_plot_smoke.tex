% Smoke fixture for log_plot snippet — verify paper loglog style key
% renders both a power-law plot and a Debye exp(-t/tau) plot at fixture
% target dimensions with Style Lock.
%
% Regression sentinel (added 2026-05-04): Plot B carries a `title=` while
% Plot A does not. Both pass `paper loglog={3.6cm}{2.6cm}`. With the style's
% `scale only axis` keyword in effect, both plots render with IDENTICAL
% inner data-area height — the title grows the bbox upward, not inward.
% If `scale only axis` is removed, Plot B's inner area shrinks vertically
% to make room for the title, breaking visual parity. This is the exact
% regression that surfaced on golden_trap_depth_picture Row 1 in 2026-05-04
% (see polymer-paper-preamble.sty `paper loglog/.style` comment block).
\documentclass[border=4pt]{standalone}
\usepackage{polymer-paper-preamble}  % loads pgfplots + paper loglog style key

\input{../../../styles/snippets/log_plot.snippet.tex}  % docs/convention only

\begin{document}
\begin{tikzpicture}[x=1cm, y=1cm]

  % Plot A: power-law I(t) ∝ t^{-n}
  \begin{scope}
    \begin{loglogaxis}[
      paper loglog={3.6cm}{2.6cm},
      xmin=1e-3, xmax=1e3, ymin=1e-4, ymax=1e1,
