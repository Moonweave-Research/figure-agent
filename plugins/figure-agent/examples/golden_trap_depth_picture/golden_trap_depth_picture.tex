% \DIBhex{x_center}{y_center}: m-DIB residue as ball-and-stick benzene ring +
% two horizontal pendant balls (chain-extension direction). Ring radius 0.22cm,
% 6 black ball vertices, pendant offset 0.18 cm. Span left-pendant to right-pendant
% = 0.80 cm. Approximates m-disubstituted benzene linearized for chain depiction.
\newcommand{\DIBhex}[2]{%
  \begin{scope}[shift={(#1,#2)}]
    \def\DIBR{0.22}%
    \foreach \k in {0,...,5} {%
      \pgfmathsetmacro{\hxa}{\DIBR*cos(\k*60)}%
      \pgfmathsetmacro{\hya}{\DIBR*sin(\k*60)}%
      \pgfmathsetmacro{\hxb}{\DIBR*cos((\k+1)*60)}%
      \pgfmathsetmacro{\hyb}{\DIBR*sin((\k+1)*60)}%
      \draw[black!85, line width=0.45pt] (\hxa,\hya) -- (\hxb,\hyb);}
    \foreach \k in {0,...,5} {%
      \pgfmathsetmacro{\hxc}{\DIBR*cos(\k*60)}%
      \pgfmathsetmacro{\hyc}{\DIBR*sin(\k*60)}%
      \node[circle, fill=black!85, draw=black, line width=0.18pt,
            inner sep=0pt, minimum size=1.20mm] at (\hxc,\hyc) {};}
    \def\DIBpen{0.18}%
    \draw[black!85, line width=0.42pt] (-\DIBR,0) -- (-\DIBR-\DIBpen,0);
    \node[circle, fill=black!85, draw=black, line width=0.18pt,
          inner sep=0pt, minimum size=1.10mm] at (-\DIBR-\DIBpen,0) {};
    \draw[black!85, line width=0.42pt] (\DIBR,0) -- (\DIBR+\DIBpen,0);
    \node[circle, fill=black!85, draw=black, line width=0.18pt,
          inner sep=0pt, minimum size=1.10mm] at (\DIBR+\DIBpen,0) {};
  \end{scope}
}

% \InvVulcChain{x_phase}{y_anchor}: linear S/DIB inverse-vulcanization copolymer.
% Pattern (chemistry-faithful, scope-relative):
%   DIB_1 @ x=0.40 — S_3 — DIB_2 @ x=2.16 — S_6 (trap site) — DIB_3 @ x=4.64 — S_3 trail.
% DIB residues = ball-and-stick m-benzene with 2 horizontal pendants (-CH2-...-S-).
% S_n runs = amber pearl chain, C-S transition bonds drawn in black to mark
% organic-sulfur boundary. Total span ~5.8 cm. S_6 middle run = trap-site cluster.
\newcommand{\InvVulcChain}[2]{%
  \begin{scope}[shift={(#1,#2)}]
    %% DIB residues
    \DIBhex{0.40}{0}
    \DIBhex{2.16}{0}
    \DIBhex{4.64}{0}
    %% C-S boundary bond: DIB1 right pendant (0.80) → S_3 first atom (1.04)
    \draw[black!75, line width=0.42pt] (0.80,0) -- (1.04,0);
    %% S_3 left run
    \foreach \i in {1,...,3} {%
      \pgfmathsetmacro{\sx}{0.80+\i*0.24}%
      \draw[cAmber, line width=1.05pt, line cap=round]
        (\sx-0.24,0) -- (\sx,0);
      \node[circle, fill=cAmber, draw=cAmber!75!black, line width=0.32pt,
            inner sep=0pt, minimum size=1.65mm] at (\sx,0) {};}
    %% S_3 last (1.52) → DIB2 left pendant (1.76)
    \draw[black!75, line width=0.42pt] (1.52,0) -- (1.76,0);
    %% DIB2 right pendant (2.56) → S_6 first atom (2.80)
    \draw[black!75, line width=0.42pt] (2.56,0) -- (2.80,0);
    %% S_6 middle run (RICH — dashed-box highlighted)
    \foreach \i in {1,...,6} {%
      \pgfmathsetmacro{\sx}{2.56+\i*0.24}%
      \draw[cAmber, line width=1.05pt, line cap=round]
        (\sx-0.24,0) -- (\sx,0);
      \node[circle, fill=cAmber, draw=cAmber!75!black, line width=0.32pt,
            inner sep=0pt, minimum size=1.65mm] at (\sx,0) {};}
    %% S_6 last (4.00) → DIB3 left pendant (4.24)
    \draw[black!75, line width=0.42pt] (4.00,0) -- (4.24,0);
    %% DIB3 right pendant (5.04) → S_3 first atom (5.28)
    \draw[black!75, line width=0.42pt] (5.04,0) -- (5.28,0);
    %% S_3 right run (chain trail; no closing DIB)
    \foreach \i in {1,...,3} {%
      \pgfmathsetmacro{\sx}{5.04+\i*0.24}%
      \draw[cAmber, line width=1.05pt, line cap=round]
        (\sx-0.24,0) -- (\sx,0);
      \node[circle, fill=cAmber, draw=cAmber!75!black, line width=0.32pt,
            inner sep=0pt, minimum size=1.65mm] at (\sx,0) {};}
  \end{scope}
}
